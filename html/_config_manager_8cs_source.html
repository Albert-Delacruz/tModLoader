354  public static bool EnumerableEquals(IEnumerable a, IEnumerable b) {
355  IEnumerator enumeratorA = a.GetEnumerator();
356  IEnumerator enumeratorB = b.GetEnumerator();

358  bool hasNextB = enumeratorB.MoveNext();
